import numpy as np
import pandas as pd
from scipy.fft import rfft, rfftfreq

class FeatureEngineeringPipeline:
    """
    Vectorized mathematical and statistical feature extraction pipeline 
    optimized for high-frequency cryptocurrency time-series.
    """
    
    @staticmethod
    def calculate_hurst(series: pd.Series, window: int = 300, lag_min: int = 2, lag_max: int = 20) -> pd.Series:
        """
        Estimates the rolling Hurst Exponent using Ernie Chan's variance of differences method.
        H < 0.5: Mean Reverting, H = 0.5: Random Walk, H > 0.5: Trending.
        """
        def _single_hurst(window_slice):
            if len(window_slice) < window:
                return np.nan
            lags = np.arange(lag_min, lag_max)
            # Volatility sampled at intervals of lag tau: std(z(t) - z(t - tau))
            tau = [np.std(np.subtract(window_slice[lag:], window_slice[:-lag])) for lag in lags]
            # Linear fit on log-log coordinates: log(tau) vs log(std_dev)
            poly = np.polyfit(np.log(lags), np.log(tau), 1)
            return poly  # The slope represents the Hurst Exponent

        return series.rolling(window=window).apply(_single_hurst, raw=True)

    @staticmethod
    def calculate_volume_zscore(volume: pd.Series, window: int = 20) -> pd.Series:
        """
        Normalizes volume into a rolling Z-score to extract anomalous trading volume.
        """
        rolling_mean = volume.rolling(window=window).mean()
        rolling_std = volume.rolling(window=window).std()
        # Prevent division by zero anomalies
        rolling_std = rolling_std.replace(0, np.nan).fillna(method='bfill')
        return (volume - rolling_mean) / rolling_std

    @staticmethod
    def calculate_order_book_imbalance(dataframe: pd.DataFrame, levels: int = 5) -> pd.Series:
        """
        Computes multi-level order book imbalance (OBI) bounded between -1.0 and 1.0.
        Expects order book columns formatted as: 'bid_sz_00', 'ask_sz_00', etc.
        """
        bid_cols = [f"bid_sz_{i:02d}" for i in range(levels)]
        ask_cols = [f"ask_sz_{i:02d}" for i in range(levels)]
        
        # Ensure columns exist in dataframe; if not, return neutral imbalance
        if not all(col in dataframe.columns for col in bid_cols + ask_cols):
            return pd.Series(0.0, index=dataframe.index)
            
        sum_bids = dataframe[bid_cols].sum(axis=1)
        sum_asks = dataframe[ask_cols].sum(axis=1)
        
        denominator = sum_bids + sum_asks
        imbalance = (sum_bids - sum_asks) / denominator
        return imbalance.fillna(0.0)

    @staticmethod
    def calculate_dominant_frequency(series: pd.Series, window: int = 128, sampling_rate: float = 1.0) -> pd.Series:
        """
        Applies a rolling Fast Fourier Transform (FFT) to extract the dominant cycle frequency.
        """
        def _get_dominant_freq(window_slice):
            n = len(window_slice)
            if n < window:
                return np.nan
            # Fast Fourier Transform of real-valued signal
            fourier = rfft(window_slice)
            frequencies = rfftfreq(n, d=1.0/sampling_rate)
            magnitudes = np.abs(fourier)
            # Suppress the DC offset (frequency = 0)
            magnitudes = 0.0
            dominant_index = np.argmax(magnitudes)
            return frequencies[dominant_index]

        return series.rolling(window=window).apply(_get_dominant_freq, raw=True)

def populate_any_indicators(dataframe: pd.DataFrame, metadata: dict, strategy) -> pd.DataFrame:
    """
    Main Freqtrade indicator entry point.
    All training features must be prefixed with '%' to be isolated by FreqAI.
    """
    pipeline = FeatureEngineeringPipeline()
    
    # Mathematical and Spectral Features
    dataframe['%-hurst'] = pipeline.calculate_hurst(dataframe['close'], window=300)
    dataframe['%-vol_zscore'] = pipeline.calculate_volume_zscore(dataframe['volume'], window=20)
    dataframe['%-dominant_fft_freq'] = pipeline.calculate_dominant_frequency(dataframe['close'], window=128)
    
    # Microstructural Features
    dataframe['%-order_book_imbalance_l5'] = pipeline.calculate_order_book_imbalance(dataframe, levels=5)
    
    return dataframe

The following script implements the complete, vectorized Triple-Barrier labeling pipeline:
import numpy as np
import pandas as pd

class TripleBarrierLabeler:
    """
    Implements Marcos Lopez de Prado's path-aware Triple-Barrier labeling method,
    dynamically scaled by exponential standard deviation volatility.
    """
    
    @staticmethod
    def get_dynamic_volatility(close: pd.Series, span: int = 100) -> pd.Series:
        """
        Calculates exponential moving daily volatility (returns standard deviation).
        """
        returns = close.pct_change()
        vol = returns.ewm(span=span, adjust=False).std()
        return vol.fillna(method='bfill')

    @staticmethod
    def apply_triple_barrier(
        close: pd.Series,
        volatility: pd.Series,
        pt_sl: tuple[float, float] = (2.0, 1.0),
        vertical_horizon: int = 24
    ) -> pd.DataFrame:
        """
        Computes barrier touches and returns labels:
        1.0 : Take-profit hit first
        -1.0: Stop-loss hit first
        0.0 : Time barrier reached first
        """
        labels = pd.DataFrame(index=close.index, columns=["t1", "bin"])
        timestamps = close.index
        pt_mult, sl_mult = pt_sl
        
        for i in range(len(close) - 1):
            start_time = timestamps[i]
            vol = volatility.iloc[i]
            if pd.isna(vol) or vol <= 0:
                continue
                
            start_price = close.iloc[i]
            upper_barrier = start_price * (1.0 + pt_mult * vol)
            lower_barrier = start_price * (1.0 - sl_mult * vol)
            
            # Vertical barrier index limit
            end_idx = min(i + vertical_horizon, len(close) - 1)
            future_prices = close.iloc[i+1 : end_idx+1]
            future_times = timestamps[i+1 : end_idx+1]
            
            # Default assignment is time expiry (vertical barrier)
            labels.at[start_time, "t1"] = timestamps[end_idx]
            labels.at[start_time, "bin"] = 0.0
            
            # Track price trajectory within the horizon
            for t, price in zip(future_times, future_prices):
                if price >= upper_barrier:
                    labels.at[start_time, "t1"] = t
                    labels.at[start_time, "bin"] = 1.0
                    break
                elif price <= lower_barrier:
                    labels.at[start_time, "t1"] = t
                    labels.at[start_time, "bin"] = -1.0
                    break
                    
        return labels

def set_freqai_targets(dataframe: pd.DataFrame, metadata: dict, strategy) -> pd.DataFrame:
    """
    Sets prediction targets for the FreqAI machine learning model.
    All targets must begin with '&' to be detected by FreqAI.
    """
    labeler = TripleBarrierLabeler()
    vol = labeler.get_dynamic_volatility(dataframe['close'], span=100)
    
    # Calculate path-aware Triple-Barrier binary labels
    tbm_df = labeler.apply_triple_barrier(dataframe['close'], vol, pt_sl=(2.0, 1.0), vertical_horizon=24)
    
    dataframe['&-s_tbm_bin'] = tbm_df['bin'].astype(float)
    return dataframe

import numpy as np
import pandas as pd
from hmmlearn.hmm import GaussianHMM
from sklearn.svm import OneClassSVM

class AnomalyFilteringEngine:
    """
    Implements high-fidelity anomaly detection and market regime classification.
    """
    def __init__(self, n_components: int = 2, nu: float = 0.05):
        self.n_components = n_components
        self.nu = nu
        self.hmm = GaussianHMM(n_components=self.n_components, covariance_type="full", max_iter=10000, random_state=100)
        self.oc_svm = OneClassSVM(kernel="rbf", nu=self.nu, gamma="scale")
        
    def fit_regimes(self, returns: pd.Series) -> np.ndarray:
        """
        Trains a Gaussian HMM on historical log returns to identify the active regime.
        """
        clean_returns = returns.dropna().values.reshape(-1, 1)
        self.hmm.fit(clean_returns)
        hidden_states = self.hmm.predict(clean_returns)
        
        # Sort states so that State 0 always represents the lowest volatility regime
        state_means = [np.mean(clean_returns[hidden_states == i]) for i in range(self.n_components)]
        state_vars = [np.var(clean_returns[hidden_states == i]) for i in range(self.n_components)]
        
        ordered_regimes = np.argsort(state_vars)
        mapped_states = np.array([np.where(ordered_regimes == state) for state in hidden_states])
        return mapped_states

    def fit_outliers(self, features: np.ndarray) -> None:
        """
        Fits a One-Class Support Vector Machine decision boundary on standardized features.
        """
        self.oc_svm.fit(features)
        
    def predict_outlier(self, features: np.ndarray) -> np.ndarray:
        """
        Returns 1.0 for in-distribution data points, and -1.0 for anomalies.
        """
        return self.oc_svm.predict(features)

import numpy as np
import pandas as pd

class RiskManagementEngine:
    """
    Implements a fractional Kelly and ATR position-sizing framework 
    with dynamic leverage scaling.
    """
    def __init__(self, win_rate: float, win_loss_ratio: float, kelly_fraction: float = 0.25):
        self.w = win_rate
        self.r = win_loss_ratio
        self.gamma = kelly_fraction  # e.g., 0.25 for Quarter-Kelly

    def get_fractional_kelly_risk(self) -> float:
        """
        Calculates the safe capital risk fraction.
        """
        if self.r <= 0 or self.w <= 0:
            return 0.0
        f_star = self.w - ((1.0 - self.w) / self.r)
        fractional_risk = f_star * self.gamma
        return max(fractional_risk, 0.0)

    def calculate_atr_distance(self, high: pd.Series, low: pd.Series, close: pd.Series, period: int = 14, multiplier: float = 2.0) -> float:
        """
        Calculates the Average True Range (ATR) stop distance.
        """
        tr1 = high - low
        tr2 = (high - close.shift(1)).abs()
        tr3 = (low - close.shift(1)).abs()
        
        tr = pd.concat([tr1, tr2, tr3], axis=1).max(axis=1)
        atr = tr.rolling(window=period).mean().iloc[-1]
        return atr * multiplier

    def compute_allocation(self, account_balance: float, entry_price: float, stop_distance: float) -> dict:
        """
        Calculates position size and leverage for the trade.
        """
        risk_fraction = self.get_fractional_kelly_risk()
        if risk_fraction <= 0 or stop_distance <= 0:
            return {"units": 0.0, "exposure": 0.0, "leverage": 0.0}
            
        allowed_loss = account_balance * risk_fraction
        position_units = allowed_loss / stop_distance
        total_exposure = position_units * entry_price
        leverage = total_exposure / account_balance
        
        return {
            "units": position_units,
            "exposure": total_exposure,
            "leverage": leverage,
            "risk_pct": risk_fraction * 100.0
        }

        stop_distance = risk_manager.calculate_atr_distance(
            dataframe['high'], dataframe['low'], dataframe['close'], 
            period=14, multiplier=2.0
        )
        
        account_balance = self.wallets.get_total_stake_amount()
        
        # Compute capital allocation using the unified sizing formula
        allocation = risk_manager.compute_allocation(
            account_balance=account_balance, 
            entry_price=current_rate, 
            stop_distance=stop_distance
        )
        
        target_stake = allocation["exposure"]
        
        # Restrict stake size to framework parameters
        if target_stake < min_stake:
            return 0.0
        return min(target_stake, max_stake)
